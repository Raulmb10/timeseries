# Convert to DataFrame
df = pd.DataFrame(data)

# --- Part a: Plot the two time series ---
plt.figure(figsize=(12, 6))
plt.plot(df['GS3M'], label='3-Month Treasury Bill Rate (GS3M)', color='blue')

plt.ylabel('Interest Rate (%)')

plt.grid()
plt.show()

# --- Part b: Unit Root Test (ADF) ---
def adf_test(series, title=''):
    result = adfuller(series.dropna(), autolag='AIC')
    print(f'ADF Test for {title}:')
    print(f'ADF Statistic: {result[0]}')
    print(f'p-value: {result[1]}')
    print(f'Critical Values: {result[4]}')
    if result[1] < 0.05:
        print(f'{title} is stationary (reject null)\n')
    else:
        print(f'{title} is non-stationary (fail to reject null)\n')

# Test levels
adf_test(df['GS3M'], 'GS3M (Levels)')
adf_test(df['GS6M'], 'GS6M (Levels)')

# Test first differences
adf_test(df['GS3M'].diff().dropna(), 'ΔGS3M (First Differences)')
adf_test(df['GS6M'].diff().dropna(), 'ΔGS6M (First Differences)')

# --- Part c: Cointegration Test ---
# Engle-Granger cointegration test
score, p_value, _ = coint(df['GS3M'], df['GS6M'])
print('Cointegration Test (Engle-Granger):')
print(f'Test Statistic: {score}')
print(f'p-value: {p_value}')
if p_value < 0.05:
    print('GS3M and GS6M are cointegrated (reject null)\n')
else:
    print('GS3M and GS6M are not cointegrated (fail to reject null)\n')

# --- Part e: VAR Model (if cointegrated, use levels; if not, use differences) ---
# Assuming cointegration (based on economic intuition), use levels
model = VAR(df)
results = model.fit(4)  # 4 lags
print('VAR Model Summary (4 lags, levels):')
print(results.summary())
