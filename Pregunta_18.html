import matplotlib.pyplot as plt
from statsmodels.tsa.stattools import adfuller

# Time series plot of the data
plt.figure(figsize=(12, 6))

plt.xlabel('Time')

plt.title('Time Series Plot of Log DPI')
plt.grid(True)
plt.show()

# Test for stationarity using the Augmented Dickey-Fuller test
result = adfuller(df[1])
print('ADF Statistic:', result[0])
print('p-value:', result[1])

# Apply differencing if necessary
if result[1] > 0.05:  # If p-value > 0.05, the series is likely non-stationary
    df['diff_1'] = df[1].diff(periods=1)
    df['diff_2'] = df['diff_1'].diff(periods=1)

    # Plot the differenced series
    plt.figure(figsize=(12, 6))
    plt.plot(df[0], df['diff_1'])
    plt.xlabel('Time')
    plt.ylabel('Differenced Log DPI (1st order)')
    plt.title('Time Series Plot of Differenced Log DPI (1st order)')
    plt.grid(True)
    plt.show()

    plt.figure(figsize=(12, 6))
    plt.plot(df[0], df['diff_2'])
    plt.xlabel('Time')
    plt.ylabel('Differenced Log DPI (2nd order)')
    plt.title('Time Series Plot of Differenced Log DPI (2nd order)')
    plt.grid(True)
    plt.show()

    # Perform ADF test on differenced series
    result_diff_1 = adfuller(df['diff_1'].dropna())

    result_diff_2 = adfuller(df['diff_2'].dropna())

import pandas as pd
import numpy as np
from statsmodels.tsa.arima.model import ARIMA
from statsmodels.stats.diagnostic import acorr_ljungbox
from scipy.stats import jarque_bera
from statsmodels.graphics.tsaplots import plot_acf
import matplotlib.pyplot as plt

# Prepare data
data = df['log_diff_1'].dropna()

# Define ARIMA orders to try (example)
orders = [(0, 1, 0), (0, 1, 1), (1, 1, 0), (1, 1, 1), (1, 1, 2), (2, 1, 1)]
results = []

for order in orders:
    try:
        # Fit ARIMA model
        model = ARIMA(data, order=order)
        fitted_model = model.fit()

        # Evaluate model
        aic = fitted_model.aic
        bic = fitted_model.bic
        rmse = np.sqrt(np.mean(fitted_model.resid ** 2))

        # Diagnostic tests
        lb_test = acorr_ljungbox(fitted_model.resid, lags=[10], return_df=True)
        lb_pvalue = lb_test['lb_pvalue'][10]  # Access p-value correctly
        jb_test = jarque_bera(fitted_model.resid)
        jb_pvalue = jb_test[1]

        # Store results
        results.append([order, aic, bic, rmse, lb_pvalue, jb_pvalue])

    except Exception as e:
        print(f"Error fitting ARIMA model with order {order}: {e}")
        results.append([order, np.nan, np.nan, np.nan, np.nan, np.nan])

# Create summary table
